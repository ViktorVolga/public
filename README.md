# public
helper repo for youtube videos

Создаем Докерфайл
sudo docker build . -t public
    . - докер будет искать докерфайл для сборки образа в текущей папке - можно передать и другой путь
    -t public - устанавливает тэг для образа
sudo docker images 
    Просмотр доступных образов docker

sudo docker run -p 2223:22 -d --name public -it -v $(pwd):/opt/ public-image /bin/bash
    -p 2223:22 
        пробрасываем локальный порт 2223 на порт 22 контейнера
    -d 
        Запускает контейнер в виде демона не блокируя терминал
    --name public-container 
        имя с которым создается контейнер
    -it
        -i
            интерактивный режим - позволит перенаправить ввод (stdin) в контейнер если захотим в будущем зайти в него
        -t
            (tty) - создает псевдотерминал
    -v $(pwd):/opt/
        примонтирует указанную локальную папку (в нашем случае $(pwd) возвращает путь до текущей папки) к папке /opt/ внутри контейнера
    public-image 
        - тэг запускаемого образа - можно заменить на id
        /bin/bash - запускаем шэл внутри контейнера

sudo docker ps
    - выводит список запущенных контейнров

sudo docker ps -a
    - выводит список всех доступных локально контейнеров

sudo docker exec public-container /usr/sbin/sshd
    Запускаем внутри контейнера sshd

ssh root@localhost -p 2223
    Пробуем приконеткится к контейнеру
